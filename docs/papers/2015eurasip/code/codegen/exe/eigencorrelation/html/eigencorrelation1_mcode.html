 4%% Output
 5% (R) correlation matrix 
 6% (E) eigenvalues diagonal matrix
 7% (V) eigenvectors 
 8% (M) largest eigenvalue
 9function [R,E,V,M] = eigencorrelation(X)
10    numLines = size(X,1);
11    numColumns = size(X,2);
12    Y = zeros(numLines,numColumns);
13    
14    for i = 1:numLines
15        Vx = X(i,:);
16        stdv = std(Vx,1);
17        if stdv > 0          
18            Y(i,:) = (Vx - mean(Vx)) / stdv;    % standardization, have zero mean and unit variance
19        end
20    end
21    
22    R = 1/numColumns*Y*Y';                              % estimation of correlation matrix (R)
23    [V,E] = eig(R);                                     % eigenvectors (V) and eigenvalues (E) of correlation matrix
24    M(1) = max(diag(E));                                % largest eigenvalue (M)
25    
26    Ed  = diag(E);    
27    for i = 1:size(Ed,1);
28        if M(1) == Ed(i)
29            M(2) = i;                                   % get the position of largest eigenvalue
30        end        
31    end
